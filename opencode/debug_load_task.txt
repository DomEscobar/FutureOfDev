    const taskFile = path.join(BENCHMARK_TASKS_PATH, `${taskId}.json`);
    console.log('[DEBUG] Looking for task file:', taskFile);
    console.log('[DEBUG] BENCHMARK_TASKS_PATH:', BENCHMARK_TASKS_PATH);

    }

    console.log('[DEBUG] Exception loading task:', e.message);

}
